Return the most frequently occurring review site.

    PREFIX qpr: <http://istresearch.com/qpr> 
    SELECT ?review_site  (count(?ad) AS ?count)(group_concat(?ad;separator=',') AS ?ads)
    WHERE { 
        ?ad a qpr:Ad ;
        qpr:location 'Fayetteville, AR' ; 
        qpr:review_site ?review_site . 
    } 
         GROUP BY ?review_site
         ORDER BY DESC(?count)
         LIMIT 1
